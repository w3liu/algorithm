// 循环方式
func binary(arr []int, v int) int {
	l, h := 0, len(arr)-1
	for l <= h {
		mid := l + (h-l)>>1
		if arr[mid] == v {
			return mid
		}
		if arr[mid] > v {
			h = mid - 1
		}
		if arr[mid] < v {
			l = mid + 1
		}
	}
	return -1
}
// 递归方式
func binary1(arr []int, v int) int {
	return binaryIndex(arr, 0, len(arr)-1, v)
}

func binaryIndex(arr []int, l, h, v int) int {
	if l > h {
		return -1
	}
	mid := l + (h-l)>>1
	if arr[mid] == v {
		return mid
	}
	if arr[mid] < v {
		return binaryIndex(arr, l+1, h, v)
	} else {
		return binaryIndex(arr, l, h-1, v)
	}
}
